15// 生成随机 Hex 字符串
16static void generate_random_hex(char* buffer, size_t length)
17{
18 static const char hex_chars[] = "0123456789abcdef";
19
20 for (size_t i = 0; i < length - 1; i++) {
21 buffer[i] = hex_chars[rand() % 16];
22 }
23 buffer[length - 1] = '\0';
24}

40// 解析 W3C Trace Context 头
41// 格式：traceparent: 00-<trace-id>-<span-id>-<flags>
42int parse_traceparent(const char* header, char* trace_id, char* parent_span_id, int* sampled)
43{
44 if (!header || strlen(header) != 55) {
45 return -1;
46 }
47
48 // 验证版本
49 if (strncmp(header, "00-", 3) != 0) {
50 return -1;
51 }
52
53 // 提取 trace-id (位置 3-34)
54 strncpy(trace_id, header + 3, 32);
55 trace_id[32] = '\0';
56
57 // 提取 parent-span-id (位置 35-50)
58 strncpy(parent_span_id, header + 35, 16);
59 parent_span_id[16] = '\0';
60
61 // 提取 flags (位置 52-53)
62 int flags = (int)strtol(header + 52, NULL, 16);
63 *sampled = (flags & 0x01);
64
65 return 0;
66}

77// 判断是否应该采样
78static int should_sample(void)
79{
80 double sample_rate = g_gateway_config.observability.tracing_sample_rate;
81
82 // 采样率 1.0 = 100% 采样
83 if (sample_rate >= 1.0) return 1;
84 if (sample_rate <= 0.0) return 0;
85
86 return ((double)rand() / RAND_MAX) < sample_rate;
87}

91// 在连接建立时初始化追踪上下文
92void tracing_init_context(client_ctx_t* ctx, const char* incoming_traceparent)
93{
94 if (!g_gateway_config.observability.enable_tracing) {
95 return;
96 }
97
98 // 生成 Request ID（用于日志关联）
99 generate_random_hex(ctx->request_id, sizeof(ctx->request_id));
100
101 if (incoming_traceparent && parse_traceparent(incoming_traceparent,
102 ctx->trace_id,
103 ctx->span_id,
104 &ctx->is_sampled) == 0) {
105 // 从上游继承追踪上下文
106 log_debug(ctx, "trace_context_inherited", "Inherited trace context from upstream");
107 } else {
108 // 创建新的追踪
109 generate_trace_id(ctx->trace_id, sizeof(ctx->trace_id));
110 generate_span_id(ctx->span_id, sizeof(ctx->span_id));
111 ctx->is_sampled = should_sample();
112
113 if (ctx->is_sampled) {
114 log_debug(ctx, "trace_context_created", "Created new trace context");
115 }
116 }
117}

153// 导出 Span
154void tracing_export_span(client_ctx_t* ctx, const char* operation, double duration_ms)
155{
156 if (!g_gateway_config.observability.enable_tracing || !ctx->is_sampled) {
157 return;
158 }
159
160 const char* exporter = g_gateway_config.observability.tracing_exporter;
161
162 if (strcmp(exporter, "console") == 0) {
163 export_to_console(ctx->trace_id, ctx->span_id, operation, duration_ms);
164 } else if (strcmp(exporter, "jaeger") == 0 ||
165 strcmp(exporter, "zipkin") == 0 ||
166 strcmp(exporter, "otlp") == 0) {
167 export_to_http_collector(ctx->trace_id, ctx->span_id, operation, duration_ms);
