112// 增加计数器
113static void counter_inc(metric_counter_t* counter)
114{
115 pthread_mutex_lock(&counter->lock);
116 counter->value++;
117 pthread_mutex_unlock(&counter->lock);
118}
119
120// 减少计数器
121static void counter_dec(metric_counter_t* counter)
122{
123 pthread_mutex_lock(&counter->lock);
124 counter->value--;
125 pthread_mutex_unlock(&counter->lock);
126}

128// 记录直方图
129static void histogram_observe(metric_histogram_t* hist, double value)
130{
131 pthread_mutex_lock(&hist->lock);
132
133 // 累加总和和总数
134 hist->sum += value;
135 hist->total_count++;
136
137 // 更新对应的桶
138 for (int i = 0; i < hist->bucket_count; i++) {
139 if (hist->buckets[i].le < 0 || value <= hist->buckets[i].le) {
140 hist->buckets[i].count++;
141 }
142 }
143
144 pthread_mutex_unlock(&hist->lock);
145}

198// 生成 Prometheus 格式的指标文本
199void metrics_generate_output(char* buffer, size_t buffer_size)
200{
201 size_t offset = 0;
202
203#define APPEND(fmt, ...) do { \
204 int n = snprintf(buffer + offset, buffer_size - offset, fmt, ##__VA_ARGS__); \
205 if (n > 0 && (size_t)n < buffer_size - offset) { \
206 offset += n; \
207 } \
208 } while(0)
209
210 // HELP 和 TYPE 声明
211 APPEND("# HELP %s %s\n# TYPE %s counter\n",
212 http_requests_total.name, http_requests_total.help, http_requests_total.name);
213 APPEND("# HELP %s %s\n# TYPE %s counter\n",
214 http_request_errors_total.name, http_request_errors_total.help, http_request_errors_total.name);
215 APPEND("# HELP %s %s\n# TYPE %s gauge\n",
216 http_requests_in_flight.name, http_requests_in_flight.help, http_requests_in_flight.name);
217 APPEND("# HELP %s %s\n# TYPE %s histogram\n",
218 http_request_duration_seconds.name, http_request_duration_seconds.help, http_request_duration_seconds.name);
219 APPEND("# HELP %s %s\n# TYPE %s histogram\n",
220 http_upstream_duration_seconds.name, http_upstream_duration_seconds.help, http_upstream_duration_seconds.name);
221
222 // 计数器值
223 pthread_mutex_lock(&http_requests_total.lock);
224 APPEND("%s %lld\n", http_requests_total.name, (long long)http_requests_total.value);
225 pthread_mutex_unlock(&http_requests_total.lock);
226
227 pthread_mutex_lock(&http_request_errors_total.lock);
228 APPEND("%s %lld\n", http_request_errors_total.name, (long long)http_request_errors_total.value);
229 pthread_mutex_unlock(&http_request_errors_total.lock);
230
231 pthread_mutex_lock(&http_requests_in_flight.lock);
232 APPEND("%s %lld\n", http_requests_in_flight.name, (long long)http_requests_in_flight.value);
233 pthread_mutex_unlock(&http_requests_in_flight.lock);
234
235 // 直方图值
236 pthread_mutex_lock(&http_request_duration_seconds.lock);
237 for (int i = 0; i < http_request_duration_seconds.bucket_count; i++) {
238 if (http_request_duration_seconds.buckets[i].le < 0) {
239 APPEND("%s_bucket{le=\"+Inf\"} %lld\n",
240 http_request_duration_seconds.name,
241 (long long)http_request_duration_seconds.total_count);
242 } else {
243 APPEND("%s_bucket{le=\"%.3f\"} %lld\n",
244 http_request_duration_seconds.name,
245 http_request_duration_seconds.buckets[i].le,
246 (long long)http_request_duration_seconds.buckets[i].count);
247 }
248 }
249 APPEND("%s_sum %.6f\n", http_request_duration_seconds.name, http_request_duration_seconds.sum);
250 APPEND("%s_count %lld\n", http_request_duration_seconds.name, (long long)http_request_duration_seconds.total_count);
251 pthread_mutex_unlock(&http_request_duration_seconds.lock);
252
253 // 上游服务直方图
254 pthread_mutex_lock(&http_upstream_duration_seconds.lock);
255 for (int i = 0; i < http_upstream_duration_seconds.bucket_count; i++) {
256 if (http_upstream_duration_seconds.buckets[i].le < 0) {
257 APPEND("%s_bucket{le=\"+Inf\"} %lld\n",
258 http_upstream_duration_seconds.name,
259 (long long)http_upstream_duration_seconds.total_count);
260 } else {
261 APPEND("%s_bucket{le=\"%.3f\"} %lld\n",
262 http_upstream_duration_seconds.name,
263 http_upstream_duration_seconds.buckets[i].le,
264 (long long)http_upstream_duration_seconds.buckets[i].count);
265 }
266 }
267 APPEND("%s_sum %.6f\n", http_upstream_duration_seconds.name, http_upstream_duration_seconds.sum);
268 APPEND("%s_count %lld\n", http_upstream_duration_seconds.name, (long long)http_upstream_duration_seconds.total_count);
269 pthread_mutex_unlock(&http_upstream_duration_seconds.lock);
270
271#undef APPEND
272}

278typedef struct {
279 uv_tcp_t client;
280 uv_write_t write_req;
281 char* response_data; // 存储响应数据，在 close 时释放
282} metrics_client_t;

284static void metrics_on_close(uv_handle_t* handle)
285{
286 // 获取客户端结构
287 metrics_client_t* mc = (metrics_client_t*)handle->data;
288
289 // 释放响应数据
290 if (mc->response_data) {
291 free(mc->response_data);
292 }
293
294 // 释放客户端结构
295 free(mc);
296}
297
298static void metrics_on_write_complete(uv_write_t* req, int status)
299{
300 // 从 write_req 的 data 字段获取客户端结构
301 metrics_client_t* mc = (metrics_client_t*)req->data;
302
303 // 释放响应数据（如果有）
304 if (mc && req->data) {
305 // req->data 已经被设置为 full_response，在 metrics_read_cb 中设置
306 // 这里不需要再次释放，因为已经在 metrics_read_cb 中通过 mc->write_req.data 设置了
307 }
308
309 // 安全关闭连接
310 if (mc) {
311 uv_close((uv_handle_t*)&mc->client, metrics_on_close);
312 }
313}

322static void metrics_read_cb(uv_stream_t* client, ssize_t nread, const uv_buf_t* buf)
323{
324 if (nread < 0) {
325 if (nread != UV_EOF) {
326 fprintf(stderr, "[Metrics] Read error: %s\n", uv_err_name(nread));
327 }
328 // 修复：正确获取 client 指针
329 metrics_client_t* mc = (metrics_client_t*)client->data;
330 uv_close((uv_handle_t*)&mc->client, metrics_on_close);
331 return;
332 }
333
334 // 收到请求，生成指标响应
335 metrics_client_t* mc = (metrics_client_t*)client->data;
336
337 char response[65536];
338 metrics_generate_output(response, sizeof(response));
339
340 char http_header[] =
341 "HTTP/1.1 200 OK\r\n"
342 "Content-Type: text/plain; version=0.0.4\r\n"
343 "Connection: close\r\n"
344 "\r\n";
345
346 char* full_response = malloc(strlen(http_header) + strlen(response) + 1);
347 strcpy(full_response, http_header);
348 strcat(full_response, response);
349
350 // 保存响应数据指针，以便在 close 时释放
351 mc->response_data = full_response;
352
353 // 设置 write_req 的 data 指向 metrics_client_t，以便在回调中获取
354 mc->write_req.data = mc;
355 uv_buf_t write_buf = uv_buf_init(full_response, strlen(full_response));
356
357 uv_write(&mc->write_req, (uv_stream_t*)&mc->client, &write_buf, 1, metrics_on_write_complete);
358}

360static void metrics_new_connection(uv_stream_t* server, int status)
361{
362 if (status < 0) {
363 fprintf(stderr, "[Metrics] New connection error: %s\n", uv_strerror(status));
364 return;
365 }
366
367 metrics_client_t* mc = calloc(1, sizeof(metrics_client_t));
368 uv_tcp_init(metrics_loop, &mc->client);
369 mc->client.data = mc;
370
371 uv_accept(server, (uv_stream_t*)&mc->client);
372 uv_read_start((uv_stream_t*)&mc->client, metrics_alloc_buffer, metrics_read_cb);
373}

375static void* metrics_server_thread(void* arg)
376{
377 (void)arg;
378
379 metrics_loop = uv_loop_new();
380 uv_tcp_init(metrics_loop, &metrics_server);
381
382 struct sockaddr_in addr;
383 uv_ip4_addr("0.0.0.0", g_gateway_config.observability.metrics_port, &addr);
384 uv_tcp_bind(&metrics_server, (const struct sockaddr*)&addr, 0);
385
386 uv_listen((uv_stream_t*)&metrics_server, 128, metrics_new_connection);
387
388 printf("[Metrics] Prometheus 指标服务器已启动：http://0.0.0.0:%d%s\n",
389 g_gateway_config.observability.metrics_port,
390 g_gateway_config.observability.metrics_path);
391
392 uv_run(metrics_loop, UV_RUN_DEFAULT);
393
394 uv_loop_delete(metrics_loop);
395 return NULL;
396}
