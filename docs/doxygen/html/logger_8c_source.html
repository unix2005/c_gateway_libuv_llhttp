13// 获取当前时间戳（ISO 8601 格式）
14static void get_timestamp_iso(char *buffer, size_t size)
15{
16 struct timeval tv;
17 gettimeofday(&tv, NULL);
18
19 time_t now = tv.tv_sec;
20 struct tm *tm_info = localtime(&now);
21
22 char time_buffer[32];
23 strftime(time_buffer, sizeof(time_buffer), "%Y-%m-%dT%H:%M:%S", tm_info);
24
25 // 修复：使用 int 类型，微秒范围是 0-999999，除以 1000 后是 0-999
26 int milliseconds = (int)(tv.tv_usec / 1000);
27 if (milliseconds < 0)
28 milliseconds += 1000;
29 snprintf(buffer, size, "%s.%03dZ", time_buffer, milliseconds);
30}

33uint64_t get_time_nanoseconds(void)
34{
35 struct timespec ts;
36 clock_gettime(CLOCK_MONOTONIC, &ts);
37 return (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;
38}

58// 转义 JSON 字符串中的特殊字符
59static void escape_json_string(const char *input, char *output, size_t output_size)
60{
61 size_t j = 0;
62 for (size_t i = 0; input[i] && j < output_size - 2; i++)
63 {
64 char c = input[i];
65 if (c == '"' || c == '\\')
66 {
67 if (j < output_size - 3)
68 {
69 output[j++] = '\\';
70 output[j++] = c;
71 }
72 }
73 else if (c == '\n')
74 {
75 if (j < output_size - 3)
76 {
77 output[j++] = '\\';
78 output[j++] = 'n';
79 }
80 }
81 else if (c == '\r')
82 {
83 if (j < output_size - 3)
84 {
85 output[j++] = '\\';
86 output[j++] = 'r';
87 }
88 }
89 else if (c == '\t')
90 {
91 if (j < output_size - 3)
92 {
93 output[j++] = '\\';
94 output[j++] = 't';
95 }
96 }
97 else
98 {
99 output[j++] = c;
100 }
101 }
102 output[j] = '\0';
103}

105// 核心日志函数 - 结构化 JSON 格式
106void log_request(log_level_t level, client_ctx_t *ctx,
107 const char *event, const char *format, ...)
108{
109 // 检查日志开关和级别
110 if (!g_gateway_config.observability.enable_logging ||
111 level < g_gateway_config.observability.log_level)
112 {
113 return;
114 }
115
116 // 获取时间戳
117 char timestamp[64];
118 get_timestamp_iso(timestamp, sizeof(timestamp));
119
120 // 构建日志消息
121 char message[1024];
122 va_list args;
123 va_start(args, format);
124 vsnprintf(message, sizeof(message), format, args);
125 va_end(args);
126
127 // 转义消息
128 char escaped_message[sizeof(message) * 2];
129 escape_json_string(message, escaped_message, sizeof(escaped_message));
130
131 FILE *log_file = stdout;
132 if (strlen(g_gateway_config.log_path) > 0)
133 {
134 log_file = fopen(g_gateway_config.log_path, "a");
135 if (!log_file)
136 {
137 log_file = stdout;
138 }
139 }
140
141 if (g_gateway_config.observability.enable_json_log)
142 {
143 // JSON 格式输出
144 fprintf(log_file,
145 "{"
146 "\"timestamp\":\"%s\","
147 "\"level\":\"%s\","
148 "\"service\":\"gateway\","
149 "\"event\":\"%s\"",
150 timestamp,
151 log_level_str(level),
152 event);
153
154 // 添加请求上下文（如果有）
155 if (ctx)
156 {
157 char escaped_req_id[128];
158 char escaped_trace_id[128];
159 escape_json_string(ctx->request_id, escaped_req_id, sizeof(escaped_req_id));
160 escape_json_string(ctx->trace_id, escaped_trace_id, sizeof(escaped_trace_id));
161
162 fprintf(log_file,
163 ",\"request_id\":\"%s\","
164 "\"trace_id\":\"%s\","
165 "\"span_id\":\"%s\"",
166 escaped_req_id,
167 escaped_trace_id,
168 ctx->span_id);
169 }
170
171 // 添加消息
172 fprintf(log_file, ",\"message\":\"%s\"}\n", escaped_message);
173 }
174 else
175 {
176 // 传统文本格式（兼容旧日志）
177 if (ctx)
178 {
179 fprintf(log_file, "[%s] [%s] [req=%s] [trace=%s] %s: %s\n",
180 timestamp,
181 log_level_str(level),
182 ctx->request_id,
183 ctx->trace_id,
184 event,
185 message);
186 }
187 else
188 {
189 fprintf(log_file, "[%s] [%s] %s: %s\n",
190 timestamp,
191 log_level_str(level),
192 event,
193 message);
194 }
195 }
196
197 if (log_file != stdout && log_file != stderr)
198 {
199 fclose(log_file);
200 }
201}
