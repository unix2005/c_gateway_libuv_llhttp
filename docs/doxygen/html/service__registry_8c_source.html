13// IP 地址解析（支持 IPv4、IPv6 和域名）
14int parse_ip_address(const char* host, ip_address_t* addr)
15{
16 // 初始化
17 memset(addr, 0, sizeof(ip_address_t));
18
19 if (!host || strlen(host) == 0) {
20 return -1;
21 }
22
23 // 保存原始 host（可能是域名）
24 strncpy(addr->address, host, sizeof(addr->address) - 1);
25
26 // 检测是否为域名（包含字母）
27 int is_domain = 0;
28 for (size_t i = 0; i < strlen(host); i++) {
29 if ((host[i] >= 'a' && host[i] <= 'z') ||
30 (host[i] >= 'A' && host[i] <= 'Z')) {
31 is_domain = 1;
32 break;
33 }
34 }
35
36 addr->is_domain = is_domain;
37
38 // 如果是域名，尝试 DNS 解析获取 IP
39 if (is_domain) {
40 struct addrinfo hints, *res;
41 memset(&hints, 0, sizeof(hints));
42 hints.ai_family = AF_UNSPEC; // 允许 IPv4 或 IPv6
43 hints.ai_socktype = SOCK_STREAM;
44
45 int error = getaddrinfo(host, NULL, &hints, &res);
46 if (error != 0) {
47 fprintf(stderr, "[DNS] 无法解析主机：%s\n", host);
48 // 即使解析失败，也保留域名，后续由 libcurl 处理
49 addr->is_ipv6 = 0;
50 return 0; // 不返回错误，允许使用域名
51 }
52
53 // 检查是否为 IPv6
54 if (res->ai_family == AF_INET6) {
55 addr->is_ipv6 = 1;
56 struct sockaddr_in6* ipv6_addr = (struct sockaddr_in6*)res->ai_addr;
57 inet_ntop(AF_INET6, &ipv6_addr->sin6_addr, addr->address, sizeof(addr->address));
58 printf("[DNS] 解析 %s -> IPv6: %s\n", host, addr->address);
59 } else {
60 addr->is_ipv6 = 0;
61 struct sockaddr_in* ipv4_addr = (struct sockaddr_in*)res->ai_addr;
62 inet_ntop(AF_INET, &ipv4_addr->sin_addr, addr->address, sizeof(addr->address));
63 printf("[DNS] 解析 %s -> IPv4: %s\n", host, addr->address);
64 }
65
66 freeaddrinfo(res);
67 } else {
68 // 直接 IP 地址，检测 IPv4/IPv6
69 addr->is_ipv6 = (strchr(host, ':') != NULL) ? 1 : 0;
70 printf("[IP] 直接使用%s地址：%s\n",
71 addr->is_ipv6 ? "IPv6" : "IPv4", host);
72 }
73
74 return 0;
75}

171{
172 pthread_mutex_lock(&g_registry.lock);
173
174 for (int i = 0; i < g_registry.service_count; i++) {
175 service_t* svc = &g_registry.services[i];
176

178 pthread_mutex_lock(&svc->lock);
179
180 for (int j = 0; j < svc->instance_count; j++) {
181 if (strcmp(svc->instances[j].host, host) == 0 &&
182 svc->instances[j].port == port) {
183 // 删除实例（移动数组元素）
184 for (int k = j; k < svc->instance_count - 1; k++) {
185 svc->instances[k] = svc->instances[k + 1];

187 svc->instance_count--;
188
189 pthread_mutex_unlock(&svc->lock);
190 pthread_mutex_unlock(&g_registry.lock);
191
192 printf("[Gateway] 服务实例注销：%s -> %s:%d\n", name, host, port);
193 return 0;
194 }
195 }
196
197 pthread_mutex_unlock(&svc->lock);
198 }
199 }

201 pthread_mutex_unlock(&g_registry.lock);

225service_instance_t* service_select_instance(service_t* service)
226{
227 if (!service || service->instance_count == 0) {
228 return NULL;
229 }
230
231 pthread_mutex_lock(&service->lock);
232
233 // 轮询负载均衡
234 int start = service->current_instance;
235 service_instance_t* selected = NULL;
236
237 do {
238 service_instance_t* inst = &service->instances[service->current_instance];
239 service->current_instance = (service->current_instance + 1) % service->instance_count;
240
241 if (inst->health != SERVICE_UNHEALTHY) {
242 selected = inst;
243 inst->request_count++;
244 break;
245 }
246 } while (service->current_instance != start);
247
248 pthread_mutex_unlock(&service->lock);
249
250 if (selected) {
251 const char* proto_str = (selected->protocol == PROTOCOL_HTTPS) ? "HTTPS" : "HTTP";
252 const char* ip_ver = selected->ip_addr.is_ipv6 ? "IPv6" : "IPv4";
253 printf("[Gateway] 选择服务实例：%s -> [%s] %s:%d (%s) [请求数：%d]\n",
254 service->name, ip_ver, selected->host, selected->port,
255 proto_str, selected->request_count);
256 }
257
258 return selected;
259}
