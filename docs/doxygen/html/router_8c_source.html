89void send_response(client_ctx_t *client, int status_code, const char *content_type, char *body_to_send)
90{
91 // 1. 创建写入上下文
92 write_ctx_t *wctx = malloc(sizeof(write_ctx_t));
93 memset(wctx, 0, sizeof(write_ctx_t));
94
95 // 2. 准备 Header
96 // 我们必须 malloc Header，因为 send_response 返回后栈上的局部变量会失效
97 wctx->header_ptr = malloc(512);
98 int header_len = sprintf(wctx->header_ptr,
99 "HTTP/1.1 %d OK\r\n"
100 "Content-Type: %s\r\n"
101 "Content-Length: %zu\r\n"
102 "Connection: keep-alive\r\n"
103 "\r\n",
104 status_code, content_type, body_to_send ? strlen(body_to_send) : 0);
105
106 // 3. 关联 Body
107 // 假设 body_to_send 是由 cJSON_Print 生成的堆指针
108 wctx->body_ptr = body_to_send;
109
110 // 4. 设置 uv_buf_t 指向这些堆内存
111 wctx->bufs[0] = uv_buf_init(wctx->header_ptr, header_len);
112 int nbufs = 1;
113
114 if (wctx->body_ptr)
115 {
116 wctx->bufs[1] = uv_buf_init(wctx->body_ptr, strlen(wctx->body_ptr));
117 nbufs = 2;
118 }
119
120 // 5. 启动异步写入，绑定回调
121 int r = uv_write(&wctx->req, (uv_stream_t *)&client->handle, wctx->bufs, nbufs, on_write_completed);
122
123 if (r < 0)
124 {
125 fprintf(stderr, "uv_write failed immediately: %s\n", uv_strerror(r));
126 // 如果启动失败，需要手动清理，因为回调不会被触发
127 free(wctx->header_ptr);
128 if (wctx->body_ptr)
129 free(wctx->body_ptr);
130 free(wctx);
131 }
132}
